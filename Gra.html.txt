<script>
// Tu lógica original (ajustada mínimamente para que corra)
const precios    = { '250': 4000, '500': 8000, '1000': 16000 };
const cantidades = { '250': 0, '500': 0, '1000': 0 };
const cocos      = { '250': 'con coco', '500': 'con coco', '1000': 'con coco' };
const WHATSAPP_NUMBER = '542914619955';

function cambiar(tam, delta) {
  cantidades[tam] = Math.max(0, cantidades[tam] + delta);
  document.getElementById('qty-' + tam).textContent = cantidades[tam];
  const card = document.getElementById('card-' + tam);
  const row  = document.getElementById('coco-row-' + tam);
  
  if (cantidades[tam] > 0) {
    card.classList.add('activo');
    if(row) row.classList.add('visible');
  } else {
    card.classList.remove('activo');
    if(row) row.classList.remove('visible');
  }
  actualizarTotal();
}

function setCoco(tam, v) {
  cocos[tam] = v;
  document.getElementById('coco-' + tam + '-con').classList.toggle('on', v === 'con coco');
  document.getElementById('coco-' + tam + '-sin').classList.toggle('on', v === 'sin coco');
}

function actualizarTotal() {
  let t = 0;
  for (const k in cantidades) t += cantidades[k] * precios[k];
  document.getElementById('totalDisplay').innerHTML = '<span>$</span>' + t.toLocaleString('es-AR');
}

function enviar() {
  const nombre   = document.getElementById('nombre').value.trim();
  const apellido = document.getElementById('apellido').value.trim();
  if (!nombre || !apellido) { alert('Por favor ingresá tu nombre y apellido.'); return; }
  
  const labels = { '250': '250 g', '500': '500 g', '1000': '1 kg' };
  const lineas = [];
  let total = 0;
  
  for (const k in cantidades) {
    if (cantidades[k] > 0) {
      const sub = cantidades[k] * precios[k];
      total += sub;
      lineas.push(`• ${cantidades[k]}x Granola ${labels[k]} (${cocos[k]}) – $${sub.toLocaleString('es-AR')}`);
    }
  }
  
  if (!lineas.length) { alert('Agregá al menos un producto.'); return; }
  
  const msg = `Hola Homey! 🌾 Quiero hacer un pedido:\n\nNombre: ${nombre} ${apellido}\n\n${lineas.join('\n')}\n\nTotal: $${total.toLocaleString('es-AR')}`;
  window.open('https://wa.me/' + WHATSAPP_NUMBER + '?text=' + encodeURIComponent(msg), '_blank');
  
  document.getElementById('gracias').style.display = 'block';
}

['nombre','apellido'].forEach(id => {
  document.getElementById(id).addEventListener('input', () => {
    document.getElementById('gracias').style.display = 'none';
  });
});
